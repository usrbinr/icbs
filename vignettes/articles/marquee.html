Marquee Integration for Colored Labels

Overview

The marquee package provides rich text formatting for R graphics, including the ability to color text inline. The stwd package integrates with marquee to create labels where the text color automatically matches data values. ## Why Colored Labels?

In data visualizations, you often want labels to match their meaning:

  • Green text for positive variances
  • Red text for negative variances

This creates visual consistency and immediate comprehension.

The marquee_color_labels() Function

This function formats text labels with marquee color syntax:

# Basic usage
labels <- c("Good", "Bad", "Neutral")
colors <- c("#006400", "#8B0000", "#404040")

marquee_color_labels(labels, colors)
[1] "{#006400 Good}"    "{#8B0000 Bad}"     "{#404040 Neutral}"

With Formatting Options

# Bold labels
marquee_color_labels(labels, colors, bold = TRUE)
[1] "{#006400 **Good**}"    "{#8B0000 **Bad**}"     "{#404040 **Neutral**}"
# Italic labels
marquee_color_labels(labels, colors, italic = TRUE)
[1] "{#006400 *Good*}"    "{#8B0000 *Bad*}"     "{#404040 *Neutral*}"
# Bold and italic
marquee_color_labels(labels, colors, bold = TRUE, italic = TRUE)
[1] "{#006400 ***Good***}"    "{#8B0000 ***Bad***}"    
[3] "{#404040 ***Neutral***}"

Using with ggplot2 and marquee

Manual Approach

You can use marquee_color_labels() to pre-format your data, then use marquee::geom_marquee():

library(marquee)

#  Define colors for examples
cols <- list(actual = "#404040", variance_positive = "#006400", variance_negative = "#8B0000", previous = "#A0A0A0")

# Prepare data
variance_data <- data.frame(
    category = c("Product A", "Product B", "Product C", "Product D"),
    value = c(120, 85, 105, 95),
    variance = c(20, -5, 5, -5)
)

# Determine colors based on variance
variance_data$text_color <- ifelse(
    variance_data$variance >= 0,
    cols$variance_positive,
    cols$variance_negative
)

# Create formatted labels
variance_data$formatted_label <- marquee_color_labels(
    sprintf("%+d", variance_data$variance),
    variance_data$text_color,
    bold = TRUE
)

# Plot with marquee
ggplot(variance_data, aes(x = category, y = value)) +
    geom_col(fill = cols$actual, width = 0.6) +
    geom_marquee(
        aes(label = formatted_label),
        vjust = -0.5,
        size = 4
    ) +
    labs(title = "Sales with Colored Variance Labels") +
    theme_minimal()

Using geom_marquee_colored()

For convenience, use geom_marquee_colored() which handles the formatting automatically:

library(marquee)

variance_data$label <- sprintf("%+d", variance_data$variance)

ggplot(variance_data, aes(x = category, y = value)) +
    geom_col(fill = cols$actual, width = 0.6) +
    geom_marquee_colored(
        data = variance_data,
        mapping = aes(x = category, y = value + 5, label = label, color = text_color),
        bold = TRUE,
        size = 4
    ) +
    labs(title = "Sales with Auto-Colored Variance Labels") +
    theme_minimal()

Practical Examples

Variance Chart with Colored Labels

library(marquee)

#  Define colors for examples
cols <- list(actual = "#404040", variance_positive = "#006400", variance_negative = "#8B0000", previous = "#A0A0A0")

# Sales performance data
sales <- data.frame(
    region = c("North", "South", "East", "West", "Central"),
    actual = c(150, 120, 180, 95, 110),
    plan = c(140, 130, 170, 100, 100)
)

sales$variance <- sales$actual - sales$plan
sales$variance_pct <- round((sales$variance / sales$plan) * 100, 1)
sales$var_color <- ifelse(sales$variance >= 0,
                          cols$variance_positive,
                          cols$variance_negative)
sales$label <- sprintf("%+.1f%%", sales$variance_pct)

ggplot(sales, aes(x = reorder(region, variance))) +
    # Reference bars (plan)
    geom_col(aes(y = plan), fill = NA, color = cols$actual,
             linewidth = 0.8, width = 0.6) +
    # Actual bars
    geom_col(aes(y = actual), fill = cols$actual, width = 0.4) +
    # Colored variance labels
    geom_marquee_colored(
        data = sales,
        mapping = aes(
            x = reorder(region, variance),
            y = actual + 8,
            label = label,
            color = var_color
        ),
        bold = TRUE,
        size = 3.5
    ) +
    coord_flip() +
    labs(
        title = "Regional Sales Performance vs Plan",
        x = NULL, y = "Sales ($M)"
    ) +
    theme_minimal()

Waterfall with Colored Value Labels

library(marquee)

#  Define colors for examples
cols <- list(actual = "#404040", variance_positive = "#006400", variance_negative = "#8B0000", previous = "#A0A0A0")

# Bridge data
bridge <- data.frame(
    driver = c("2023", "Volume", "Price", "Mix", "FX", "2024"),
    value = c(100, 15, 8, -5, -3, NA),
    type = c("start", "positive", "positive", "negative", "negative", "end")
)

# Calculate positions for waterfall
bridge$running <- cumsum(ifelse(is.na(bridge$value), 0, bridge$value))
bridge$running[nrow(bridge)] <- bridge$running[nrow(bridge) - 1]
bridge$ymin <- c(0, head(bridge$running, -1))
bridge$ymax <- bridge$running
bridge$ymin[bridge$type %in% c("start", "end")] <- 0

# Determine colors
bridge$bar_fill <- ifelse(
    bridge$type %in% c("start", "end"),
    cols$actual,
    cols$actual_light
)
bridge$label_color <- ifelse(
    bridge$type %in% c("start", "end"),
    "white",
    ifelse(bridge$value >= 0, cols$variance_positive, cols$variance_negative)
)

# Create labels
bridge$label <- ifelse(
    bridge$type %in% c("start", "end"),
    sprintf("%.0f", bridge$ymax),
    sprintf("%+.0f", bridge$value)
)

bridge$driver <- factor(bridge$driver, levels = bridge$driver)

ggplot(bridge, aes(x = driver)) +
    geom_rect(
        aes(xmin = as.numeric(driver) - 0.4,
            xmax = as.numeric(driver) + 0.4,
            ymin = ymin, ymax = ymax),
        fill = bridge$bar_fill,
        color = cols$actual
    ) +
    geom_marquee_colored(
        data = bridge,
        mapping = aes(
            x = driver,
            y = (ymin + ymax) / 2,
            label = label,
            color = label_color
        ),
        bold = TRUE,
        size = 4
    ) +
    labs(title = "Revenue Bridge with Colored Labels", x = NULL, y = "Revenue ($M)") +
    theme_minimal()

Marquee Syntax Reference

The marquee package uses a markdown-like syntax for formatting:

Syntax Result
{.red text} Red colored text
{#FF0000 text} Hex-colored text
**text** Bold text
*text* Italic text
***text*** Bold italic text
{.red **text**} Bold red text

Installation

If marquee is not installed, install it from CRAN:

install.packages("marquee")

Best Practices

  1. Use sparingly - Colored text is powerful; don’t overuse it

  2. Match semantics - Green for good, red for bad (in Western contexts)

  3. Ensure readability - Make sure colors have sufficient contrast

  4. Consider accessibility - Some users may have color vision deficiencies; use shape and position as additional indicators

  5. Be consistent - Use the same color meanings throughout your report
